# 문자열(3)  
{: .no_toc }

```c++
#include <iostream>
#include <stdio.h>
#include <wchar.h>

//글자 개수가 음수가 될리없음
unsigned int GetLen(const wchar_t* _pStr)
//함수가 받아와야 하는것은 무엇일까....? 바로 문자열의 주소..!
{
    int i = 0; // 반복문 안에서 선언하면 알죠? 계속 선언되어서 룻됩니다.
    while (true) // True 딱히 조건찾기 애매할때 좋음
    {
        wchar_t c = *(_pStr + i);


        if ('\0' == c) //NULL 문자 0와 같음 참고로 상수를 먼저 적고 변수와 비교가 좋다(실수 방지)
        {
            break;
        }
        ++i;
    }
    return i;
}

//양수만 GL(수정않되게 와이드캐릭터 포인터 (불러들인 값))
unsigned int GL(const wchar_t* PointerString)
{
    int i = 0;
    while ('\0' != PointerString[i])
    {
        i++;
    }
    return i;
}

int main()
{
    wchar_t szName[10] = L"Raidmond";

    int iLen = wcslen(szName); //기본 제공된 함수
    int iLen2 = GetLen(szName); //선생님의 함수
    int iLen3 = GL(szName); // 내가 만들어본 함수
    return 0;
}
```
